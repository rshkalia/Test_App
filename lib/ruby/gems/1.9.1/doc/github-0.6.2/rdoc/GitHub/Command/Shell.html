def run
  GitHub.debug "sh: #{command}"

  out = err = nil
  Open3.popen3(*@command) do |_, pout, perr|
    out = pout.read.strip
    err = perr.read.strip
  end

  replace @error = err unless err.empty?
  replace @out = out unless out.empty?

  self
end
