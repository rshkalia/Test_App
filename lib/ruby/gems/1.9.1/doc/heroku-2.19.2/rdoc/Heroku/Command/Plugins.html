def install
  plugin = Heroku::Plugin.new(args.shift)
  if plugin.install
    begin
      Heroku::Plugin.load_plugin(plugin.name)
    rescue Exception => ex
      installation_failed(plugin, ex.message)
    end
    display "#{plugin} installed"
  else
    error "Could not install #{plugin}. Please check the URL and try again"
  end
end
